An example where threads 'share' a variable:

#include <stdio.h>
#include <omp.h>
int main()
{
	int sum = 0;
#pragma omp parallel
  {
     int id = omp_get_thread_num();
     sum += 5;
     printf("Thread %d: sum = %d\n", id, sum);     
  }
	
  return 0;
}

	int sum = 0;
#pragma omp parallel
